//旋转数组的最小数字public class Solution {    public int minNumberInRotateArray(int [] array) {        if(array==null||array.length==0)            	return 0;        int start=0;        int end=array.length-1;        int mid=0;        while(array[start]>=array[end]){            if(end-start==1){                mid=end;                return array[mid];            }            mid=(start+end)/2;            if(array[start]==array[end]&&array[start]==array[mid])                return getminNumberInRotateArray(array);            if(array[mid]<=array[end])                	end=mid;            else if(array[mid]>=array[start])                	start=mid;        }        return array[mid];        }    int getminNumberInRotateArray(int[] array){        int res = array[0];        for(int i=1;i<array.length;i++){            if (array[i]<res){                res=array[i];            }        }        return res;    }}
