／／两个链表的第一个公共节点public ListNode FindFirstCommonNode(ListNode pHead1, ListNode pHead2) {        if(pHead1==null||pHead2==null)            return null;        ListNode pNode1=pHead1;        ListNode pNode2=pHead2;        int i=0;        int j=0;        while(pNode1!=null){            pNode1=pNode1.next;            i++;        }         while(pNode2!=null){            pNode2=pNode2.next;            j++;        }        pNode1=pHead1;        pNode2=pHead2;        if(i>j){            int diff=i-j;            for(int m=0;m<diff;m++)                pNode1=pNode1.next;            while(pNode1!=null&&pNode2!=null&&pNode1!=pNode2){                pNode1=pNode1.next;                pNode2=pNode2.next;            }                        }else{            int diff=j-i;            for(int m=0;m<diff;m++)                pNode2=pNode2.next;            while(pNode1!=null&&pNode2!=null&&pNode1!=pNode2){                pNode1=pNode1.next;                pNode2=pNode2.next;            }        }        return pNode1;    }
