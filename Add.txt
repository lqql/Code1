／／求1+2+3+...+n，要求不能使用乘除法、for、while、if、else、switch、case等关键字及条件判断语（A?B:C）public int Sum_Solution(int n) {         return (n*n+n)>>1;    }不用加减乘除做加法public int Add(int num1,int num2) {        int sum;        int carry;        do{            sum=num1 ^ num2;            carry=(num1 & num2)<<1;            num1=sum;            num2=carry;        }while(num2!=0);        return num1;    }
