／／在排序的数组中找到第一次出现的和为s的两个数public ArrayList<Integer> FindNumbersWithSum(int [] array,int sum) {          ArrayList<Integer> list = new ArrayList<Integer>();        if(array==null||array.length<2)            return list;        int maxMultiply=0;        int i=0;        int j=array.length-1;        while(i<j){            int target=array[i]+array[j];            if(sum==target){                                list.add(array[i]);                list.add(array[j]);                return list;            } else if(sum<target){                j--;            }else{                i++;            }                    }        return list;    }
