数字在排序数组中出现的次数public int GetNumberOfK(int [] array , int k) {        if(array==null||array.length<=0)           return 0;        int length=array.length;        int left=GetLeft(array,k,0,length-1);        int right=GetRight(array,k,0,length-1);        return right-left+1;    }    int GetRight(int[] array,int target,int start,int end){        int mid;        while(start<=end){            mid=(start+end)/2;            if(array[mid]<=target)                start=mid+1;            else                end=mid-1;        }        return start-1;    }    int GetLeft(int[] array,int target,int start,int end){        int mid;        while(start<=end){            mid=(start+end)/2;            if(array[mid]>=target)                end=mid-1;            else                start=mid+1;        }        return end+1;    }
