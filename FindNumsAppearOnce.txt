／／数组中只出现一次的数字：两个数字只出现一次，其余数字出现两次。public void FindNumsAppearOnce(int [] array,int num1[] , int num2[]) {        if(array==null||array.length<2)            return;        int orRes=0;        for(int i=0;i<array.length;i++){            orRes^=array[i];        }        int index=FindFirstBitIs1(orRes);        num1[0]=0;        num2[0]=0;        for(int i=0;i<array.length;i++){            if(IsBit1(array[i],index))                num1[0]^=array[i];            else                num2[0]^=array[i];        }    }    int FindFirstBitIs1(int num){        int index=0;        while((num&1)!=1){            num>>=1;            index++;        }        return index;    }    boolean IsBit1(int num,int indexBit){        num>>=indexBit;        if((num&1)==0)            return false;        else            return true;    }
