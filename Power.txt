//数值的整数次方public double Power(double base, int exponent) {        if(isEqual(base,0.0)&&exponent<0)            	return -1.0;        double result;        if(exponent<0){            exponent=-exponent;            result=getPower(base,exponent);            result=1.0/result;        }else{            result=getPower(base,exponent);        }        return result;              }    boolean isEqual(double num1,double num2){        if(num1-num2<0.0000001&&num1-num2>-0.0000001){            return true;        }        return false;    }    double getPower(double base,int exponent){        if(exponent==0)            return 1;        if(exponent==1)            return base;        double result;        result=getPower(base,exponent/2);        result*=result;        if(exponent%2 == 1)            result=result*base;        return result;}
