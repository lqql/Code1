／／按单词逆序public String ReverseSentence(String str) {        if(str==null)            return null;        int length=str.length();        char[]a=new char[length];        a=str.toCharArray();        reverse(a,0,length-1);        int start=0;        int end=0;        for(;start<length&&end<length;){            if(a[start]==' '){                start++;                end++;            }            else if(a[end]==' '){                reverse(a,start,end-1);                start=end;                        }else{                end++;            }        }        reverse(a,start,end-1);        return String.valueOf(a);            }    void reverse(char[] a,int start,int end){                while(start<end){            char temp=a[start];            a[start]=a[end];            a[end]=temp;            start++;            end--;                    }    }
