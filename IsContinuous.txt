／／扑克牌的顺子public class Solution {    public boolean isContinuous(int [] numbers) {        if(numbers==null||numbers.length<=0)            return false;        Arrays.sort(numbers);        int numbersOfZero=0;        for(int i=0;i<numbers.length&&numbers[i]==0;i++){            numbersOfZero++;        }        int gap=0;        int start=numbersOfZero;        int end=start+1;        while(end<numbers.length){            if(numbers[start]==numbers[end])                return false;            gap+=numbers[end]-numbers[start]-1;            start=end;            end++;        }        return (gap>numbersOfZero) ? false : true;    }
