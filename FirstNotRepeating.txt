第一个只出现一次的字符public int FirstNotRepeatingChar(String str) {        if(str==null)            return -1;        HashMap<Character,Integer> hashmap=new HashMap<Character,Integer>();        int length=str.length();        for(int i=0;i<length;i++){            if(hashmap.containsKey(str.charAt(i))){                int value=hashmap.get(str.charAt(i));                hashmap.put(str.charAt(i),value+1);            }else{                hashmap.put(str.charAt(i),1);            }        }        int m=0;        for(;m<length;m++){            if(hashmap.get(str.charAt(m))==1)                return m;                                    }        return -1;            }
