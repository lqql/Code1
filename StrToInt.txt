／／把字符串转换成整数public int StrToInt(String str) {        if(str==null||str.length()==0)            return 0;        str.trim();        int res=0;        boolean flag=true;        for(int i=0;i<str.length();i++){            char c=str.charAt(i);            if(i==0&&(c=='-'||c=='+')){                flag= c=='+'?true:false;            }            else if(c>='0' && c<='9'){                if(flag&&res>Integer.MAX_VALUE)                	return Integer.MAX_VALUE;            	else if(!flag&&res<Integer.MIN_VALUE)                	return Integer.MIN_VALUE;            	res *= 10;            	res += c - '0';            }else                return 0;                    }        return flag?res:-res;            }
